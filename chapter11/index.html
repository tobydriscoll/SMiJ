using CairoMakie, Printf
using ToeplitzMatrices, LinearAlgebra, SpectralMethodsTrefethen
"p28 - eigenmodes of Laplacian on the disk (compare p22)"
function p28(N=25, M=20; version=:a)
    @assert isodd(N) && iseven(M) "Must choose odd N and even M"
    # r coordinate, ranging from -1 to 1 (N must be odd):
    N2 = div(N-1, 2)
    Dr, r = cheb(N)
    D²r = Dr^2
    D₁ = D²r[2:N2+1, 2:N2+1]
    D₂ = D²r[2:N2+1, N:-1:N2+2]
    E₁ = Dr[2:N2+1, 2:N2+1]
    E₂ = Dr[2:N2+1, N:-1:N2+2]

    # θ coordinate, ranging from 0 to 2π (M must be even):
    dθ = 2π / M
    θ = dθ * (1:M)
    M2 = div(M, 2)
    c0 = -π^2 / 3dθ^2 - 1 / 6
    col = [0.5 * (-1)^(k + 1) / sin(k * dθ / 2)^2 for k in 1:M-1]
    D²θ = Toeplitz([c0; col], [c0; col])

    # Laplacian in polar coordinates:
    R = Diagonal(1 ./ r[2:N2+1])
    J = [zeros(M2, M2) I; I zeros(M2, M2)]
    Δ = kron(D₁ + R * E₁, I(M)) + kron(D₂ + R * E₂, J) + kron(R^2, D²θ)

    # Compute eigenmodes:
    λ, W = eigen(-Δ, sortby=abs)
    λ = sqrt.(real(λ / λ[1]))

    function getmode(W, k, θ, r)
        V = reshape(real(W[:, k]), M, N2)
        V = [zeros(M) V / maximum(abs, V)]                # BC and normalize
        V = [V V[(@. mod1(M2 + (1:M), M)), N2+1:-1:1]]    # extend to r < 0
        return interp2dgrid(V, fourinterp, chebinterp, θ, r)
    end

    # Plot eigenmodes:
    r, θ = range(-1, 1, 71), range(0, π, 111)
    X = [r * cos(θ) for θ in θ, r in r]
    Y = [r * sin(θ) for θ in θ, r in r]
    zcirc = cispi.((-200:200) / 200)
    fig = Figure(size=(640, 640))
    if version == :a
        modes = [1 3; 6 10]
        for (i, mode) in pairs(modes)
            str = @sprintf("Mode %d: λ = %0.11f", mode, λ[mode])
            U = getmode(W, mode, θ, r)
            ax = Axis3(fig[i[1], i[2]]; title=str, aspect=:data)
            surface!(ax, X, Y, 1.5 .+ U;
                colormap=:redsblues, colorrange=(0.5, 2.5))
            contour!(ax, X, Y, U; levels=[0], color=:darkblue, linewidth=1)
            lines!(ax, reim(zcirc)...; color=:black, linewidth=2)
            set_lights!(ax, [])
            set_ambient_light!(ax, RGBf(0.85, 0.85, 0.85))
            hidedecorations!(ax)
            hidespines!(ax)
        end
    elseif version == :b
        modes = reshape(1:25, 5, 5)'
        for (i, mode) in pairs(modes)
            str = @sprintf("%0.4f", λ[mode])
            U = getmode(W, mode, θ, r)
            ax = Axis(fig[i[1], i[2]]; title=str, titlefont=:regular, aspect=DataAspect())
            contour!(ax, X, Y, U; levels=[0], color=:darkblue, linewidth=1)
            lines!(ax, reim(zcirc)...; color=:black, linewidth=2)
            hidedecorations!(ax)
            hidespines!(ax)
        end
    end
    return fig
end

mod1

The most noteworthy bit of the program is in the getmode subfunction. To set up for interpolation on the disk using our standard functions, we need to restore the compact polar representation to all four regions depicted in Figure 11.3 of the text—specifically, to recreate the data for r<0r<0r<0. The data matrix V has θ\thetaθ along MMM rows and rrr along (N−1)/2(N-1)/2(N−1)/2 columns. To find the data for extending a row, we have to look in the row that is M/2M/2M/2 rows away, where we treat the row index circularly.

The mod1 function is perfect for working with periodic 1-indexed arrays. It computes the index modulo MMM, but with the result in 0<m≤M0<m\le M0<m≤M instead of 0≤m<M0\le m<M0≤m<M. Of course, we also have to reverse the order of the columns as we append to each row. Once that is taken care of, the interp2dgrid function doesn’t care that our grid is polar—it just sees a rectangular grid with our Fourier setup in the row dimension and Chebyshev in the column dimension.

using CairoMakie, LaTeXStrings
using ToeplitzMatrices, SpectralMethodsTrefethen
"p29 - solve Poisson equation on the unit disk (compare p16 and p28)"
function p29(N=31, M=40)
    @assert isodd(N) && iseven(M) "Must choose odd N and even M"
    # Set up r and θ:
    N2, M2 = div(N - 1, 2), div(M, 2)
    Dr, r = cheb(N)
    D²r = Dr^2
    D₁ = D²r[2:N2+1, 2:N2+1]
    D₂ = D²r[2:N2+1, N:-1:N2+2]
    E₁ = Dr[2:N2+1, 2:N2+1]
    E₂ = Dr[2:N2+1, N:-1:N2+2]
    dθ = 2π / M
    θ = dθ * (1:M)
    c0 = -π^2 / 3dθ^2 - 1 / 6
    col = [0.5 * (-1)^(k + 1) / sin(k * dθ / 2)^2 for k in 1:M-1]
    D²θ = Toeplitz([c0; col], [c0; col])

    # Laplacian in polar coordinates:
    R = Diagonal(1 ./ r[2:N2+1])
    J = [zeros(M2, M2) I; I zeros(M2, M2)]
    Δ = kron(D₁ + R * E₁, I(M)) + kron(D₂ + R * E₂, J) + kron(R^2, D²θ)

    # Right-hand side and solution for u:
    F = [-r^2 * sin(θ / 2)^4 + sin(6θ) * cos(θ / 2)^2 for θ in θ, r in r[2:N2+1]]
    v = Δ \ vec(F)

    # Reshape results, extend to r < 0, and interpolate for the plot:
    V = reshape(v, M, N2)
    V = [zeros(M) V]                               # boundary values at r = 1
    V = [V V[(@. mod1(M2 + (1:M), M)), N2+1:-1:1]]    # extend to r ∈ [-1, 0)
    r, θ = range(-1, 1, 71), range(0, π, 111)
    U = interp2dgrid(V, fourinterp, chebinterp, θ, r)
    X = [r * cos(θ) for θ in θ, r in r]
    Y = [r * sin(θ) for θ in θ, r in r]
    fig, ax, plt = contourf(X, Y, U; levels=20, colormap=:amp,
        axis=(; xlabel=L"x", ylabel=L"y", aspect=1))
    Colorbar(fig[1, 2], plt)
    return fig
end
