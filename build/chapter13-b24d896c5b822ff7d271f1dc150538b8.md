There are alternatives to the mathematical trick used here. One is similar to the approach in the upcoming [Program 33](#p33), so we will let that wait for now. Another is to modify the "boundary deletion" strategy we have used throughout the book. 

:::{code-cell}
:label: output32b
using CairoMakie, Printf, SpectralMethodsTrefethen
function p32b(N=16, bc=[1, 0])
    D, x = cheb(N)
    v = zeros(N+1)
    idxb = [1, N+1]    # boundary indices
    v[idxb] = bc
    D² = D^2
    f = (@. exp(4x[2:N])) - D²[2:N, idxb] * bc    # modify RHS for the BCs
    Δ = D²[2:N, 2:N]    # delete boundary rows and columns
    v[2:N] = Δ \ f      # solve for interior values
    fig, ax, _ = scatter(x, v)
    u = chebinterp(v)
    lines!(ax, -1..1, u)
    exact(x) = (exp(4 * x) - sinh(4) * x - cosh(4)) / 16 + (x + 1) / 2
    maxerr = maximum(abs(u(x) - exact(x)) for x in range(-1, 1, 801))
    ax.title = @sprintf("max err = %0.4e", maxerr)
    return fig
end
p32b()
